18  */

28      * @param string $name Canonical name for the object.
29      * @param UserContext $context Context within which the object is valid.

31     protected function __construct($name, UserContext $context)

53     }

55     /**
56      * Converts an array of string items to an associative array of objects of the specified type.
57      *
58      * @param array $items
59      * @param string $class
60      * @param UserContext $context
61      * @return array
62      */
63     public static function toObjectArray(array $items, $class, UserContext $context)
64     {
65         return array_combine($items, array_map(function($item) use ($class, $context) {
66             return new $class($item, $context);
67         }, $items));
68     }
69 
70     /**
71      * Converts an associative array of descriptors to objects of the specified type.
72      *
73      * @param array $items
74      * @param string $class
75      * @param UserContext $context
76      * @return array
77      */
78     public static function toRichObjectArray(array $items, $class, UserContext $context)
79     {
80         array_walk($items, function(&$value, $name) use ($class, $context) {
81             $value = new $class($name, $context, $value);
82         });
83         return $items;
84     }
85 }
